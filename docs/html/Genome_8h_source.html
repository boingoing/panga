55  size_t AddGene(size_t geneBitWidth) {
56  assert(geneBitWidth != 0);
57 
58  // First gene starts at bit index 0.
59  // Subsequent genes start at the previous index + the previous bit width.
60  size_t geneBitStartIndex = 0;
61  if (!this->_genes.empty()) {
62  const Gene& gene = this->_genes.back();
63  geneBitStartIndex = gene.startBitIndex + gene.bitWidth;
64  }
65 
66  // Align the bits underlying the gene such that it begins and ends at a byte border.
67  if (byteAlign) {
68  size_t bitGap = geneBitWidth % 8;
69  if (bitGap != 0) {
70  geneBitWidth += 8 - bitGap;
71  }
72  bitGap = geneBitStartIndex % 8;
73  if (bitGap != 0) {
74  geneBitStartIndex += 8 - bitGap;
75  }
76  }
77 
78  size_t geneIndex = this->_genes.size();
79  this->_genes.push_back({geneBitStartIndex,geneBitWidth});
80 
81  // Boolean genes are always at the end of the chromosome.
82  // Update the bit index for the start of the boolean genes.
83  this->_firstBooleanGeneBitIndex = geneBitStartIndex + geneBitWidth;
84 
85  return geneIndex;
86  }
