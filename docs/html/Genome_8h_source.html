51  size_t AddGene(size_t geneBitWidth) {
52  assert(geneBitWidth != 0);
53  assert(this->_geneStartBitIndex.size() == this->_geneBitWidth.size());
54 
55  // First gene starts at bit index 0.
56  // Subsequent genes start at the previous index + the previous bit width.
57  size_t geneBitStartIndex = this->_geneStartBitIndex.empty() ?
58  0 :
59  this->_geneStartBitIndex.back() + this->_geneBitWidth.back();
60 
61  // Align the bits underlying the gene such that it begins and ends at a byte border.
62  if (byteAlign) {
63  size_t bitGap = geneBitWidth % 8;
64  if (bitGap != 0) {
65  geneBitWidth += 8 - bitGap;
66  }
67  bitGap = geneBitStartIndex % 8;
68  if (bitGap != 0) {
69  geneBitStartIndex += 8 - bitGap;
70  }
71  }
72 
73  this->_geneStartBitIndex.push_back(geneBitStartIndex);
74  this->_geneBitWidth.push_back(geneBitWidth);
75 
76  // Boolean genes are always at the end of the chromosome.
77  // Update the bit index for the start of the boolean genes.
78  this->_firstBooleanGeneBitIndex = geneBitStartIndex + geneBitWidth;
79 
80  return this->_geneStartBitIndex.size() - 1;
81  }
