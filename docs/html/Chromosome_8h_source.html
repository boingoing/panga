181  template <typename FloatType = double, typename IntegerType = uint64_t>
182  static FloatType DecodeFloat(IntegerType integerValue, size_t bitWidth, FloatType min, FloatType max) {
183  auto localMax = std::numeric_limits<IntegerType>::max() >> (sizeof(IntegerType) * 8 - bitWidth);
184  FloatType factor = (FloatType)integerValue / (FloatType)localMax;
185  return factor * (max - min) + min;

197  template <typename FloatType = double, typename IntegerType = uint64_t>
198  static IntegerType EncodeFloat(FloatType floatVal, size_t bitWidth, FloatType min, FloatType max) {
199  auto localMax = std::numeric_limits<IntegerType>::max() >> (sizeof(IntegerType) * 8 - bitWidth);
200  FloatType factor = (floatVal - min) / (max - min);
201  return factor > 1.0 ?
202  localMax :
203  (IntegerType)(factor * localMax);

206  template <bool ignoreGeneBoundaries = true>
207  static void UniformCrossover(Chromosome* parent1, Chromosome* parent2, Chromosome* offspring, RandomWrapper* randomWrapper) {
208  assert(parent1->_bitCount == parent2->_bitCount);
209 
210  if (ignoreGeneBoundaries) {
211  for (size_t i = 0; i < parent1->_bytesCount; i++) {
212  std::byte mask = randomWrapper->RandomByte();
213  offspring->_bytes[i] = (mask & parent1->_bytes[i]) | (~mask & parent2->_bytes[i]);
214  }
215  } else {
216  const Genome* genome = parent1->GetGenome();
217  for (size_t i = 0; i < genome->GetGeneCount(); i++) {
218  Chromosome* chunkSource = randomWrapper->CoinFlip(0.5) ? parent1 : parent2;
219  size_t geneStartBitIndex = genome->GetGeneStartBitIndex(i);
220  size_t geneBitWidth = genome->GetGeneBitWitdh(i);
221  if (i >= genome->GetFirstBooleanGeneIndex()) {
222  if (chunkSource->Get(geneStartBitIndex)) {
223  offspring->Set(geneStartBitIndex);
224  } else {
225  offspring->Unset(geneStartBitIndex);
226  }
227  } else {
228  chunkSource->SubVector(offspring, geneStartBitIndex, geneStartBitIndex, geneBitWidth);
229  }
230  }
231  }
232  }

234  template <bool ignoreGeneBoundaries = true>
235  static void KPointCrossover(size_t k, Chromosome* parent1, Chromosome* parent2, Chromosome* offspring, RandomWrapper* randomWrapper) {
236  size_t left = 0;
237  size_t totalBits = ignoreGeneBoundaries ?
238  parent1->GetBitCount() :
239  parent1->_genome->GetGeneCount();
240 
241  for (size_t i = 0; i < k + 1; i++) {
242  // Copy a chunk of bits starting with left and ending at right.
243  size_t right = randomWrapper->RandomInteger(left, totalBits);
244 
245  // If this is the last chunk to take, make sure we take all the rest of the bits.
246  if (i == k) {
247  right = totalBits;
248  }
249 
250  // Decide if this chunk of bits will come from parent1 or parent2 (this alternates with each step).
251  Chromosome* chunkSource = i % 2 == 0 ? parent1 : parent2;
252 
253  // Copy the current chunk.
254  if (left < totalBits) {
255  size_t startIndex = left;
256  size_t endIndex = right;
257 
258  if (!ignoreGeneBoundaries) {
259  startIndex = parent1->_genome->GetGeneStartBitIndex(left);
260  endIndex = parent1->_genome->GetGeneStartBitIndex(right) + parent1->_genome->GetGeneBitWitdh(right);
261  }
262 
263  chunkSource->SubVector(offspring, startIndex, startIndex, endIndex - startIndex);
264  }
265 
266  left = right;
267  }
268  }
