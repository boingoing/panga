115  template <typename FloatType = double, typename IntegerType = uint64_t>
116  static FloatType DecodeFloat(IntegerType integerValue, size_t bitWidth, FloatType min, FloatType max) {
117  auto localMax = std::numeric_limits<IntegerType>::max() >> (sizeof(IntegerType) * 8 - bitWidth);
118  FloatType factor = (FloatType)integerValue / (FloatType)localMax;
119  return factor * (max - min) + min;
120  }

122  template <typename FloatType = double, typename IntegerType = uint64_t>
123  static IntegerType EncodeFloat(FloatType floatVal, size_t bitWidth, FloatType min, FloatType max) {
124  auto localMax = std::numeric_limits<IntegerType>::max() >> (sizeof(IntegerType) * 8 - bitWidth);
125  FloatType factor = (floatVal - min) / (max - min);
126  return factor > 1.0 ?
127  localMax :
128  (IntegerType)(factor * localMax);
129  }

131  template <bool ignoreGeneBoundaries = true>
132  static void UniformCrossover(Chromosome* parent1, Chromosome* parent2, Chromosome* offspring, RandomWrapper* randomWrapper) {
133  assert(parent1->_bitCount == parent2->_bitCount);
134 
135  if (ignoreGeneBoundaries) {
136  for (size_t i = 0; i < parent1->_bytesCount; i++) {
137  std::byte mask = randomWrapper->RandomByte();
138  offspring->_bytes[i] = (mask & parent1->_bytes[i]) | (~mask & parent2->_bytes[i]);
139  }
140  } else {
141  const Genome* genome = parent1->GetGenome();
142  for (size_t i = 0; i < genome->GetGeneCount(); i++) {
143  Chromosome* chunkSource = randomWrapper->CoinFlip(0.5) ? parent1 : parent2;
144  size_t geneStartBitIndex = genome->GetGeneStartBitIndex(i);
145  size_t geneBitWidth = genome->GetGeneBitWitdh(i);
146  if (i >= genome->GetFirstBooleanGeneIndex()) {
147  if (chunkSource->Get(geneStartBitIndex)) {
148  offspring->Set(geneStartBitIndex);
149  } else {
150  offspring->Unset(geneStartBitIndex);
151  }
152  } else {
153  chunkSource->SubVector(offspring, geneStartBitIndex, geneStartBitIndex, geneBitWidth);
154  }
155  }
156  }
157  }

159  template <bool ignoreGeneBoundaries = true>
160  static void KPointCrossover(size_t k, Chromosome* parent1, Chromosome* parent2, Chromosome* offspring, RandomWrapper* randomWrapper) {
161  size_t left = 0;
162  size_t totalBits = ignoreGeneBoundaries ?
163  parent1->GetBitCount() :
164  parent1->_genome->GetGeneCount();
165 
166  for (size_t i = 0; i < k + 1; i++) {
167  // Copy a chunk of bits starting with left and ending at right.
168  size_t right = randomWrapper->RandomInteger(left, totalBits);
169 
170  // If this is the last chunk to take, make sure we take all the rest of the bits.
171  if (i == k) {
172  right = totalBits;
173  }
174 
175  // Decide if this chunk of bits will come from parent1 or parent2 (this alternates with each step).
176  Chromosome* chunkSource = i % 2 == 0 ? parent1 : parent2;
177 
178  // Copy the current chunk.
179  if (left < totalBits) {
180  size_t startIndex = left;
181  size_t endIndex = right;
182 
183  if (!ignoreGeneBoundaries) {
184  startIndex = parent1->_genome->GetGeneStartBitIndex(left);
185  endIndex = parent1->_genome->GetGeneStartBitIndex(right) + parent1->_genome->GetGeneBitWitdh(right);
186  }
187 
188  chunkSource->SubVector(offspring, startIndex, startIndex, endIndex - startIndex);
189  }
190 
191  left = right;
192  }
193  }
